```javascript
const startAutoStatusFlow = async order => {
  const statusFlowMap = {
    processing: 'shipped',
    shipped: 'in_transit',
    in_transit: 'delivered',
  }
  await orderStore.updateOrderStatus(order.id, statusFlowMap[order.status])
}
```
